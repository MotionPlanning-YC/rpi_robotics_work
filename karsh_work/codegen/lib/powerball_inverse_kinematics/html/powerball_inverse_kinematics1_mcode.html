1function th_out=powerball_inverse_kinematics(T06,th_p)
  2%#codegen
  3%% Inverse Kinematics Schunk Powerball LWA 4.6
  4%   th_out=inverse_kinematics(T06, theta_previous)
  5% T06= [4x4]
  6% theta_previous= [6x1]
  7
  8% T06=zeros(4,4);
  9% th_p=[0;0;0;0;0;0];
 10% 
 11% switch nargin
 12%   case 1
 13%       T06=varargin{1};
 14%   case 2
 15%       T06=varargin{1};
 16%       th_p=varargin{2};
 17% end
 18th_ik=zeros(6,8);
 19th_limit=[170; 110; 155; 170; 140; 170;]*pi/180; %Symetric Joint limits Schunk Powerball 4.6
 20d_1=205;    %base
 21a_2=350;    %upper arm
 22d_4=305;    %forearm
 23d_6=75;         %hand
 24% th = sym('th_%d', [6 1]);
 25
 26
 27
 28
 29%%
 30% Solve for $\theta_3$
 31dx=T06(1:3,1:3)*[0;0;d_6];
 32d_elbow=T06(1:3,4)-dx-[0;0;d_1];
 33d_elbow_norm =norm(d_elbow);
 34
 35temp=pi-acos((a_2^2+d_4^2-d_elbow_norm^2)/(2*a_2*d_4));
 36
 37th_ik(3,1:4)=temp;
 38th_ik(3,5:8)=-temp;
 39
 40
 41%%
 42% Solve for $\theta_1$ & $\theta_2$
 43
 44[theta1_1,theta2_1]=subproblem2([0;0;1],[0;-1;0],[0; 0; a_2; ]+ [-d_4*sin(th_ik(3,1)); 0; d_4*cos(th_ik(3,1))],d_elbow); % 2 Solutions for theta3(1)
 45[theta1_2,theta2_2]=subproblem2([0;0;1],[0;-1;0],[0; 0; a_2; ]+[-d_4*sin(th_ik(3,5)); 0; d_4*cos(th_ik(3,5))],d_elbow);   % 2 Solutions for theta3(2)
 46
 47if isnan(theta1_1(1)) || isnan(theta1_1(2))
 48    theta1_1=[th_p(1); th_p(1)];
 49end
 50
 51if isnan(theta1_2(1)) || isnan(theta1_2(2))
 52    theta1_2=[th_p(1); th_p(1)];
 53end
 54
 55if isnan(theta2_1(1)) || isnan(theta2_1(2))
 56    theta2_1=[th_p(1); th_p(1)];
 57end
 58if isnan(theta2_2(1)) || isnan(theta2_2(2))
 59    theta2_2=[th_p(1); th_p(1)];
 60end
 61
 62%%
 63% $\theta_4$, $\theta_5$, $\theta_6$
 64th_ik(1,1:4)=[theta1_1(1:2) theta1_1(1:2)];
 65th_ik(1,5:8)=[theta1_2(1:2) theta1_2(1:2)];
 66th_ik(2,1:4)=[theta2_1(1:2) theta2_1(1:2)];
 67th_ik(2,5:8)=[theta2_2(1:2) theta2_2(1:2)];
 68
 69for z=[1:2, 5:6]
 70    th_1=th_ik(1,z);
 71    th_2=th_ik(2,z);
 72    th_3= th_ik(3,z);
 73    T01 = symDH(th_1, d_1, 0, pi/2);
 74    T12 = symDH(th_2+pi/2, 0, a_2, 0);
 75    T23 = symDH(th_3-pi/2, 0, 0, -pi/2);
 76    T02=T01*T12;
 77    T03=T02*T23;
 78    
 79    Twrist=([T03(1:3,1:3).',-T03(1:3,1:3).'*T03(1:3,4);0 0 0 1])*T06;
 80    
 81    th_ik(4,z)=atan2(-Twrist(2,3),-Twrist(1,3));
 82    th_ik(5,z)=acos(Twrist(3,3)); %
 83    th_ik(6,z)=atan2(-Twrist(3,2),Twrist(3,1));
 84    
 85    th_ik(4,z+2)=atan2(Twrist(2,3),Twrist(1,3));
 86    th_ik(5,z+2)=-acos(Twrist(3,3));   %
 87    th_ik(6,z+2)=atan2(Twrist(3,2),-Twrist(3,1));
 88end
 89
 90%%
 91% Joint limits and find closest to previous
 92counter=0;
 93temp_out=zeros(6,8);
 94for x=1:8
 95    if sum(abs(th_ik(:,x)) <=th_limit)==6
 96        counter=counter+1;
 97        temp_out(1:6,counter)=th_ik(:,x);
 98    end
 99end
100        temp_out_lim=temp_out(1:6,1:counter);
101if max(abs(th_p))>0
102    min_diff=zeros(counter,1);
103    for x=1:counter
104        % min_diff(x)=norm(temp_out_lim(:,x).'-th_p);
105        min_diff(x)=((temp_out_lim(1,x)-th_p(1))^2+(temp_out_lim(2,x)-th_p(2))^2+(temp_out_lim(3,x)-th_p(3))^2+(temp_out_lim(4,x)-th_p(4))^2+(temp_out_lim(5,x)-th_p(5))^2+(temp_out_lim(6,x)-th_p(6))^2)^.5;
106    end
107    th_out=temp_out_lim(:,min_diff==min(min_diff));
108else
109    th_out=temp_out_lim;
110end
111
112
113
114
115end
